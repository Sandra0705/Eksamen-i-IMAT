\chapter{Lektion 2}
Bevis følgende udsagn udelukkende ved brug af definitionen af konvergens for en talfølge: Hvis følgen $\{a_n\}^\infty_{n=1}$ er konvergent med $\lim_{n\rightarrow1}$ $a_n = a$, så er følgen defineret ved $bn = an + 7$ for $n \in N$ også konvergent.

\begin{defn}\textbf{Konvergens} %Ny definition
\newline
$\{a_n\}^{\infty}_{n=1}$ kaldes \textit{konvergent} med \textit{grænseværdien} $ a \in \R$ skrevet som 
\begin{align*}
    a_n \rightarrow a\  \text{for} \ n \rightarrow \infty 
\end{align*}
eller
\begin{align*}
    \lim_{n\to\infty} a_n =a
\end{align*}
Såfremt $\forall \varepsilon >0 \ \exists \ N \in \N \  \forall n \in \N : n \geq N \Rightarrow |a_n - a| < \varepsilon$ \cite[s. 47]{"ETP"}\\
\end{defn}

\subsubsection{Besvarelse}
Oplysninger:
\begin{enumerate}
    \item $\lim_{n\rightarrow1}$ $a_n = a$
    \item $bn = an + 7$
\end{enumerate}
Ud fra disse har vi påstanden at $\lim_{n\rightarrow1}$ $b_n = a+7$. Dette vil nu bevises ved at bruge definitionen på konvergens
